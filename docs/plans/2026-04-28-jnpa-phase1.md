```python
from database import get_db, get_cursor

def get_data(page=1, size=50, filters=None):
    conn = get_db()
    cur = get_cursor(conn)
    try:
        cur.execute('SELECT COUNT(*) FROM terminal_master')
        total = cur.fetchone()['count']
        cur.execute('SELECT * FROM terminal_master ORDER BY terminal_name LIMIT %s OFFSET %s',
                    [size, (page - 1) * size])
        return [dict(r) for r in cur.fetchall()], total
    finally:
        conn.close()

def save(data):
    conn = get_db()
    cur = get_cursor(conn)
    row_id = data.get('id')
    if row_id:
        cur.execute('UPDATE terminal_master SET terminal_name=%s, description=%s, is_active=%s WHERE id=%s',
                    [data['terminal_name'], data.get('description'), data.get('is_active', True), row_id])
    else:
        cur.execute('INSERT INTO terminal_master (terminal_name, description, is_active) VALUES (%s, %s, %s) RETURNING id',
                    [data['terminal_name'], data.get('description'), data.get('is_active', True)])
        row_id = cur.fetchone()['id']
    conn.commit()
    conn.close()
    return row_id

def delete(row_id):
    conn = get_db()
    cur = get_cursor(conn)
    cur.execute('DELETE FROM terminal_master WHERE id=%s', (row_id,))
    conn.commit()
    conn.close()

def get_all_active():
    conn = get_db()
    cur = get_cursor(conn)
    cur.execute('SELECT id, terminal_name FROM terminal_master WHERE is_active=TRUE ORDER BY terminal_name')
    rows = cur.fetchall()
    conn.close()
    return [dict(r) for r in rows]
```

```python
        total = cur.fetchone()['count']
```

```python
                    [size, (page - 1) * size])
        return [dict(r) for r in cur.fetchall()], total
    finally:
        conn.close()

def save(data, username=None):
    conn = get_db()
    cur = get_cursor(conn)
    # Normalize empty strings to None for date/timestamp columns
    for k in data:
        if data[k] == '':
            data[k] = None
    row_id = data.get('id')
    _computed = {'id', 'vcn_id', 'doc_status', 'created_by', 'created_at'}
    if row_id:
        cols = [k for k in data if k not in _computed]
        cur.execute(f"UPDATE expected_vessels SET {', '.join(f'{c}=%s' for c in cols)} WHERE id=%s",
                    [data[c] for c in cols] + [row_id])
    else:
        data['created_by'] = username
        cols = [k for k in data if k not in {'id'}]
        cur.execute(f"INSERT INTO expected_vessels ({', '.join(cols)}) VALUES ({', '.join(['%s']*len(cols))}) RETURNING id",
                    [data[c] for c in cols])
        row_id = cur.fetchone()['id']
    conn.commit()
    conn.close()
    return row_id

def delete(row_id):
    conn = get_db()
```

**Step 3: Create `modules/EV01/views.py`**

Key routes:
- `GET /EV01` — renders `ev01.html`
- `GET /api/EV01/data` — JSON list with pagination
- `POST /api/EV01/save` — create/update a row

```python
from flask import Blueprint, render_template, request, jsonify, session
from functools import wraps
from database import get_db, get_cursor
from . import model

bp = Blueprint('EV01', __name__, url_prefix='/EV01')

def login_required(f):
    @wraps(f)
    def decorated(*args, **kwargs):
        if 'user_id' not in session:
            return jsonify({'error': 'Not logged in'}), 401
        return f(*args, **kwargs)
    return decorated

@bp.route('/')
def index():
    return render_template('ev01.html')

@bp.route('/api/EV01/data')
@login_required
def get_data():
    page = int(request.args.get('page', 1))
    size = int(request.args.get('size', 20))
    rows, total = model.get_data(page=page, size=size)
    return jsonify({'rows': rows, 'total': total})

@bp.route('/api/EV01/save', methods=['POST'])
@login_required
def save():
    data = request.get_json() or {}
    row_id = model.save(data, username=session.get('username'))
    return jsonify({'id': row_id})

@bp.route('/api/EV01/delete', methods=['POST'])
@login_required
def delete():
    data = request.get_json() or {}
    model.delete(data['id'])
    return jsonify({'ok': True})

@bp.route('/api/EV01/move_to_vcn/<int:ev_id>', methods=['POST'])
@login_required
def move_to_vcn(ev_id):
    ev = model.get_by_id(ev_id)
    if not ev:
        return jsonify({'error': 'Record not found'}), 404
    if ev.get('vcn_id'):
        return jsonify({'error': 'Already moved to VCN'}), 400

    # Create a Draft VCN from available Expected Vessel fields
    from modules.VCN01 import model as vcn_model
    vcn_data = {
        'doc_status': 'Draft',
        'terminal_name': ev.get('terminal_name'),
        'vessel_name': ev.get('vessel_name'),
        'via_number': ev.get('via_number'),
        'loa': ev.get('loa'),
        'draft': ev.get('draft'),
        'vessel_agent_name': ev.get('agent_tank_consignee'),
        'cargo_type': ev.get('cargo_name'),
        'nor_tendered': ev.get('nor'),
        'berth_name': ev.get('berth_name'),
        'eta': ev.get('eta'),
        'ata': ev.get('ata'),
    }
    vcn_id, vcn_doc_num = vcn_model.save_header(vcn_data)
    model.mark_moved_to_vcn(ev_id, vcn_id)
    return jsonify({'vcn_id': vcn_id, 'vcn_doc_num': vcn_doc_num})
```
